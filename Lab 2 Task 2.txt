--Calculate and display each employee’s annual commission (10% of their annual salary):

--Display employees’ IDs and names who earn more than 1000 LE monthly:
SELECT SSN, Fname + ' ' + Lname AS "Full Name" FROM Employee WHERE Salary > 1000;

--Display employees’ IDs and names who earn more than 10000 LE annually:
SELECT SSN, Fname + ' ' + Lname AS "Full Name" FROM Employee WHERE Salary * 12 > 10000;

--Display the names and salaries of female employees:
SELECT Fname, Lname, Salary FROM Employee WHERE Sex = 'F';
